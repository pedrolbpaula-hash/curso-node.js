const express = require('express'); 
const app = express(); 
const port = 5000; 

app.get('/', (req, res) => {
    res.send('<h1>Página principal</h1>'); 
})

// Através desse código conseguimos ter uma pequena interação com o usuário 
app.get('/:nome', (req, res) => {
    const nomeUsuario = req.params.nome; 

    res.send('Nome do usuário: ${nomeUsuario}');
})

app.use((req, res, next) => {
    res.status(404).send('<h1>Página não encontrada</h1>');
})

app.listen(port, () => {
    console.log(`Servidor está rodando na port ${port}`); 
})

- O req recebe os dados de uma requisição, ela normalmente nos retornará como um json 
Exemplo de utilização: 

app.get('/:idade', (req, res) => {
    res.send(req.params)
    // Irá retornar a idade que a pessoa passou pela url 
})
